% stuff for the appendices:
% new command to create appendices as sections at the end of chapters
% (never used though)

\newcounter{secapp}[chapter]            % define new counter, reset on next chapter
\newcommand{\sectionappendix}[1]        % the command definition
{
  \newpage                              % appendix starts on a new page
  \renewcommand{\thesection}{\thechapter.\Alph{secapp}}
                                        % the name of the appendix, e.g. 4.B
  \stepcounter{secapp}                  % secapp++
  \refstepcounter{section}              % section++, ref=section 
  \addcontentsline{toc}{section}{\protect\numberline{\thesection}#1} 
                                        % proper line in the ToC
%  {\flushleft\Large\sffamily\bfseries\appendixname\ \thesection:\ #1\par}
                                        % appearance of appendix title
  {\flushleft\Large\sffamily\bfseries\thesection$\;\;$ #1\par}
                                        % alternative appearance of appendix title
  \sectionmark{#1}\vspace{\baselineskip}  % for the header
  \renewcommand{\thesection}{\thechapter.\arabic{section}}
                                        % restore settings for next chapter
}
